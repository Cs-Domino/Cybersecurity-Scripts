#Change local user password
Set-LocalUser -Name "username" -Password (ConvertTo-SecureString "newpassword" -AsPlainText -Force)

#Verify local
Get-LocalUser -Name "JohnDoe"


#Admin
# run as a domain admin in an elevated PowerShell session

Import-Module ActiveDirectory   # only if not already loaded

# Prompt for new password securely
$securePass = Read-Host -AsSecureString "Enter new password for DOMAIN\UserName"

# Reset the account password
Set-ADAccountPassword -Identity "DOMAIN\UserName" -Reset -NewPassword $securePass

# Optionally force user to change password at next logon
Set-ADUser -Identity "DOMAIN\UserName" -PasswordExpired $true
